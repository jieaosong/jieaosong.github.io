# helper functions
def add_to(d, k, m):
    # Add value `m` to the key `k` in a dict `d`.
    if m != 0:
        if k in d.keys():
            d[k] += m
            if d[k] == 0: d.pop(k)
        else: d[k] = m

def rotate(M):
    n = M.nrows()-1
    ans = matrix(2*n+1, 2*n+1)
    for i in range(n+1):
        for j in range(n+1):
            ans[i+j,n-i+j] += M[i,j]
    return ans

def to_fund_weight(v, typ):
    # Hardcoded function to express v in the coordinates of fundamental
    # weights, as used in LiE. `typ` can only be `B` or `D`.
    # Return the coordinates if it's dominant (i.e. all positive).
    # Otherwise return `None`.
    r = len(v)
    ans = []
    for i in range(r-1):
        c = v[i] - v[i+1]
        if c < 0: return
        else: ans.append(c)
    c = 2*v[r-1] if typ == "B" else (v[r-2] + v[r-1])
    if c < 0: return
    else: return ans + [c]

def from_fund_weight(v, typ):
    # The inverse of `to_fund_weight`.
    w = sum(v[i]*li for (i, li) in enumerate(WeylCharacterRing(typ+str(len(v))).fundamental_weights()))
    return tuple(wi for wi in w.to_vector() if wi != 0)

def to_lie(V):
    # Convert a Sage Weyl character to a LiE `pol`.
    typ = V.cartan_type().type()
    r = parent(V).rank()
    L = lie.poly_null(r)
    for (t, m) in V:
        w = to_fund_weight(t.to_vector(), typ)
        L += m*lie("X"+str(w))
    return L

def K3n(n):
    # The idea is to use exp(log(ch)) to efficiently compute just the degree-n
    # term of ch.
    # Moreover, since each piece in log(ch) is already the character of a
    # g-module, we can directly do tensor products of g-modules in LiE instead
    # of products of characters.
    V = [None]
    for k in range(1, n+1):
        V_k = lie.poly_null(12)
        for d in ZZ(k).divisors():
            # LiE does not support rational coefficients,
            # so we need to multiply by k here
            V_k += k/d*lie("X"+str([d]+[0]*11))
        V.append(lie.v_decomp(V_k, "B12"))
    # use exp to compute the degree-n term
    ans = {}
    for p in Partitions(n):
        ch = lie.poly_one(12)
        for i in p:
            ch = lie.tensor(ch, V[i], "B12")
        for (m, t) in ch.sage():
            add_to(ans, tuple(from_fund_weight(t.exponents()[0], "B")), m/prod(p)/prod(factorial(m) for m in p.to_exp()))
    return ans

def Kum(n):
    J4 = lambda n: n**4 * prod(1-1/p**4 for p in ZZ(n).prime_divisors())
    # Again we use exp(log(ch)) to compute just the degree-n term.
    # However the pieces of log(ch) are no longer characters of g-modules so
    # products have to be done in the character ring, which is slower.
    S = LaurentPolynomialRing(QQ, ["x"+str(i) for i in range(4)])
    q = PowerSeriesRing(S, default_prec=n+2).gen()
    half_x = [prod(xi**ji for (xi,ji) in zip(S.gens(), j)) for j in cartesian_product([[-1,1]] * 4) if sum(j) % 4 == 0]
    log_ch = -sum(log(1-xi**2*q**i)+log(1-xi**-2*q**i) for i in range(1, n+2) for xi in S.gens()) + sum(log(1+xxxx*q**i) for i in range(1, n+2) for xxxx in half_x)
    ans = {}
    for d in (n+1).divisors():
        for p in Partitions(d):
            ch = lie.poly_null(4)
            q, r = prod(log_ch[k]*k for k in p).quo_rem(log_ch[1])
            assert r == 0
            for (m, t) in q:
                w = to_fund_weight(1/2*vector(t.exponents()[0]), "B")
                if w: ch += m*lie("X"+str(w))
            for (m, t) in lie.v_decomp(ch, "B4").sage():
                add_to(ans, tuple(from_fund_weight(t.exponents()[0], "B")), J4((n+1)/d)*m/prod(p)/prod(factorial(m) for m in p.to_exp()))
    return ans

def hodge_diamond(d, b2, l):
    # Compute Hodge numbers of a g-module V_l for a hyperkähler manifold of
    # dimension d and second Betti number b2.
    # Examples: hodge_diamond(6, 23, [1,1]) or hodge_diamond(4, 7, [1/2]*4)
    typ = "B" if b2 % 2 == 1 else "D"
    r = b2 // 2 + 1
    R = WeylCharacterRing(typ + str(r))
    n = d / 2
    M = matrix(d+1, d+1)
    # Branch all the way to B2 to recover the Hodge numbers.
    B = [[0,0]]*(r-2)+[[1,0],[0,1]] if typ == "B" else [[0,0]]*(r-3)+[[1,0],[0,1],[0,1]]
    for (m0, t0) in lie.branch(to_lie(R(l)), 'B2', B, typ+str(r)).sage():
        for (m, t) in lie.dom_char(lie("X"+str(list(t0.exponents()[0]))), 'B2').sage():
            t = t.exponents()[0]
            p = t[0]+t[1]
            q = t[0]
            # use Hodge symmetry
            for (i,j) in set([(n+p,n+q),(n+p,n-q),(n-p,n+q),(n-p,n-q),(n+q,n+p),(n+q,n-p),(n-q,n+p),(n-q,n-p)]):
                M[i,j] += m*m0
    return M

def hodge_decomp(d, b2, l):
    # Decompose the LLV component into Hodge structures, for a very general
    # element in the moduli
    typ = "B" if b2 % 2 == 1 else "D"
    r = b2 // 2 + 1
    B = [[1 if i+1==j else 0 for i in range(r-1)] for j in range(r)]
    mul = {}
    hodge = {}
    for (m0, t0) in lie.branch(to_lie(WeylCharacterRing(typ+str(r))(l)), typ+str(r-1), B, typ+str(r)).sage():
        HS = [0]*(2*d+1)
        w = from_fund_weight(t0.exponents()[0], typ)
        D = hodge_diamond(d, b2-2, w)
        for i in range(d+1):
            for j in range(d+1):
                HS[d-i+j] += D[i,j]
        mul[w] = m0
        hodge[w] = HS
    # We have found all the pieces, it suffices to determine in which degree
    # they live.
    results = []
    def search(M, mul, bound, ans):
        if M == 0:
            results.append(ans) # record all answers
            return
        w = sorted(mul)[-1]
        for i in range(bound[w], d+1):
            if i < d and mul[w] == 1: continue
            else:
                if all(M[i,j] >= hodge[w][j] for j in range(d+1)):
                    Mi = matrix(2*d+1, 2*d+1)
                    for ii in range(2*d+1):
                        for j in range(2*d+1):
                            Mi[ii,j] = M[ii,j] if ii != i and ii != 2*d-i else M[ii,j] - hodge[w][j]
                    mul_copy = mul.copy()
                    ans_copy = ans.copy()
                    if i == d:
                        add_to(mul_copy, w, -1)
                        if w in ans_copy:
                            ans_copy[w] = tuple(sorted(list(ans_copy[w])+[d]))
                        else:
                            ans_copy[w] = (d,)
                    else:
                        add_to(mul_copy, w, -2)
                        if w in ans_copy:
                            ans_copy[w] = tuple(sorted(list(ans_copy[w])+[i,2*d-i]))
                        else:
                            ans_copy[w] = (i,2*d-i)
                    bound_copy = bound.copy()
                    bound_copy[w] = i
                    search(Mi, mul_copy, bound_copy, ans_copy)
    M = rotate(hodge_diamond(d, b2, l))
    search(M, mul, {w:0 for w in mul}, {})
    assert len(results) == 1
    return results[0]

@interact
def example(typ=("Type", selector(["K3^[n]", "Kum_n"])), n=slider(2,10,step_size=1)):
    llv = (K3n if typ == "K3^[n]" else Kum)(n)
    for (k, m) in sorted(llv.items(), reverse=True):
        print(str(k) + ": " + str(m))
        b2 = 23 if typ == "K3^[n]" else 7
        print(hodge_diamond(2*n, b2, k))
        print(hodge_decomp(2*n, b2, k))
